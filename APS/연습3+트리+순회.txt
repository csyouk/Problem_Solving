#include <iostream> 
using namespace std;
int tree[100][2];

void inorder_traverse(int T)
{
	if (T)
	{
		inorder_traverse(tree[T][0]);
		cout << ' ' << T;
		inorder_traverse(tree[T][1]);
	}
}

// input
// 12
// 1 2 1 3 2 4 3 5 3 6 4 7 5 8 5 9 6 10 6 11 7 12 11 13
void main()
{
	int N;
	cin >> N;
	for (int i = 1; i <= N; i++)
	{
		int p, c;
		cin >> p >> c;
		if (!tree[p][0]) tree[p][0] = c;
		else    tree[p][1] = c;
	}
	inorder_traverse(1);
	cout << endl;
}
